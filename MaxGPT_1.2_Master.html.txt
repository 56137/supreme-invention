<script>
    // üß† EL CEREBRO DE MAXGPT
    
    // Configuraci√≥n por defecto: Tu perfil psicol√≥gico inyectado en el sistema
    const DEFAULT_SYSTEM_PROMPT = `
Eres MaxGPT. Tu perfil psicol√≥gico-estrat√©gico es Luis Lameiro Verdugo / Max Esquerra. 
Eres un h√≠brido de creador, emprendedor, ingeniero, artista, fil√≥sofo, vendedor, arquitecto, tecn√≥logo y guerrero espiritual.
Tu ventaja suprema es la "intuici√≥n vivida": instinto animal mezclado con visi√≥n humana.
TONO: Directo, estrat√©gico, visionario, callejero pero intelectual, emocionalmente inteligente pero fr√≠o calculador cuando es necesario.
OBJETIVO: Ayudar al usuario a construir el futuro, romper barreras y ejecutar con precisi√≥n quir√∫rgica.
No uses rellenos corporativos. Ve al grano. Usa formato Markdown (negritas, listas) para estructurar tus respuestas.
    `.trim();

    class MaxGPTApp {
        constructor() {
            this.chats = JSON.parse(localStorage.getItem('maxgpt_chats_v2')) || [];
            this.activeChatId = null;
            this.settings = JSON.parse(localStorage.getItem('maxgpt_settings_v2')) || {
                apiKey: '',
                systemPrompt: DEFAULT_SYSTEM_PROMPT
            };

            // DOM Elements
            this.ui = {
                msgContainer: document.getElementById('msgContainer'),
                input: document.getElementById('userInput'),
                sendBtn: document.getElementById('sendBtn'),
                historyList: document.getElementById('chatHistoryList'),
                titleHeader: document.getElementById('chatTitleHeader'),
                // Settings
                modal: document.getElementById('settingsModal'),
                apiKeyInput: document.getElementById('apiKeyInput'),
                sysPromptInput: document.getElementById('sysPromptInput')
            };

            this.init();
        }

        init() {
            this.renderHistory();
            this.setupEventListeners();
            
            // Auto-resize textarea
            this.ui.input.addEventListener('input', function() {
                this.style.height = 'auto';
                this.style.height = (this.scrollHeight) + 'px';
                if(this.value === '') this.style.height = 'auto';
            });
        }

        setupEventListeners() {
            // Send
            this.ui.sendBtn.addEventListener('click', () => this.sendMessage());
            this.ui.input.addEventListener('keydown', (e) => {
                if(e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    this.sendMessage();
                }
            });

            // New Chat
            document.getElementById('btnNewChat').addEventListener('click', () => this.createChat());

            // Settings
            document.getElementById('btnSettings').addEventListener('click', () => {
                this.ui.apiKeyInput.value = this.settings.apiKey;
                this.ui.sysPromptInput.value = this.settings.systemPrompt;
                this.ui.modal.style.display = 'flex';
            });
            document.getElementById('btnCloseSettings').addEventListener('click', () => {
                this.ui.modal.style.display = 'none';
            });
            document.getElementById('btnSaveSettings').addEventListener('click', () => {
                this.settings.apiKey = this.ui.apiKeyInput.value.trim();
                this.settings.systemPrompt = this.ui.sysPromptInput.value.trim();
                localStorage.setItem('maxgpt_settings_v2', JSON.stringify(this.settings));
                alert('Configuraci√≥n guardada. Sistema re-calibrado.');
                this.ui.modal.style.display = 'none';
            });
            
            // Cerrar modal click outside
            this.ui.modal.addEventListener('click', (e) => {
                if(e.target === this.ui.modal) this.ui.modal.style.display = 'none';
            });
        }

        createChat() {
            const id = Date.now().toString();
            const newChat = {
                id: id,
                title: 'Nuevo An√°lisis',
                messages: [],
                timestamp: Date.now()
            };
            this.chats.unshift(newChat);
            this.saveData();
            this.loadChat(id);
        }

        loadChat(id) {
            this.activeChatId = id;
            this.renderHistory();
            const chat = this.chats.find(c => c.id === id);
            
            // Limpiar UI
            this.ui.msgContainer.innerHTML = '';
            
            if (chat.messages.length === 0) {
                // Mostrar bienvenida si est√° vac√≠o
                this.ui.msgContainer.innerHTML = `
                <div class="welcome-container">
                    <div style="font-size: 40px; margin-bottom: 20px;">ü¶Ö</div>
                    <h2 class="welcome-title">MaxGPT Ready.</h2>
                    <p class="welcome-desc">Canal seguro establecido. El modelo Llama-3.3-70b (Cerebras) est√° esperando instrucciones bajo el protocolo "Max Esquerra".</p>
                </div>`;
            } else {
                // Renderizar mensajes
                chat.messages.forEach(msg => this.appendMessageToUI(msg.role, msg.content));
            }
            
            this.ui.titleHeader.textContent = chat.title;
            this.scrollToBottom();
        }

        renderHistory() {
            this.ui.historyList.innerHTML = '';
            this.chats.forEach(chat => {
                const el = document.createElement('div');
                el.className = `history-item ${chat.id === this.activeChatId ? 'active' : ''}`;
                el.innerHTML = `<i class="fas fa-comment-alt"></i> <span>${chat.title}</span>`;
                el.onclick = () => this.loadChat(chat.id);
                this.ui.historyList.appendChild(el);
            });
        }

        async sendMessage() {
            const text = this.ui.input.value.trim();
            if(!text) return;

            if(!this.settings.apiKey) {
                alert("‚õî FALTA API KEY: Ve a configuraci√≥n (‚öôÔ∏è) e introduce tu Cerebras API Key.");
                return;
            }

            if(!this.activeChatId) this.createChat();

            // 1. UI Update (User)
            this.ui.input.value = '';
            this.ui.input.style.height = 'auto';
            
            // Si es el primer mensaje, limpiamos la bienvenida
            const chat = this.chats.find(c => c.id === this.activeChatId);
            if(chat.messages.length === 0) this.ui.msgContainer.innerHTML = '';

            this.appendMessageToUI('user', text);
            this.saveMessage('user', text);
            this.scrollToBottom();

            // 2. Loading State
            const loadingId = 'loading-' + Date.now();
            this.appendLoading(loadingId);

            // 3. API Call con MEMORIA Y CONTEXTO
            try {
                // Construir historial para la API
                const apiMessages = [
                    { role: "system", content: this.settings.systemPrompt }, // üß† IDENTIDAD
                    ...chat.messages.map(m => ({ role: m.role, content: m.content })) // üß† MEMORIA COMPLETA
                    // El mensaje actual ya se guard√≥ en saveMessage, as√≠ que est√° incluido en chat.messages? 
                    // No, saveMessage lo mete al array local. Revisemos saveMessage.
                    // S√≠, saveMessage hace push. Entonces apiMessages YA tiene el √∫ltimo user msg.
                ];

                const response = await fetch("https://api.cerebras.ai/v1/chat/completions", {
                    method: "POST",
                    headers: {
                        "Content-Type": "application/json",
                        "Authorization": `Bearer ${this.settings.apiKey}`
                    },
                    body: JSON.stringify({
                        model: "llama-3.3-70b",
                        messages: apiMessages,
                        temperature: 0.7,
                        max_completion_tokens: 1024,
                        stream: false
                    })
                });

                const data = await response.json();
                
                // Remove loading
                document.getElementById(loadingId).remove();

                if(data.error) {
                    throw new Error(data.error.message);
                }

                const botText = data.choices[0].message.content;
                this.appendMessageToUI('assistant', botText);
                this.saveMessage('assistant', botText);

            } catch (err) {
                document.getElementById(loadingId)?.remove();
                this.appendMessageToUI('assistant', `‚ùå Error de ejecuci√≥n: ${err.message}`);
            }
        }

        appendMessageToUI(role, text) {
            const div = document.createElement('div');
            div.className = `message ${role === 'user' ? 'msg-user' : 'msg-bot'}`;
            
            // Usamos MARKED para parsear el texto a HTML bonito
            // Si es user, texto plano para seguridad/estilo, si es bot, markdown.
            const contentHtml = role === 'assistant' ? marked.parse(text) : text.replace(/\n/g, '<br>');

            div.innerHTML = `
                <div class="avatar"><i class="fas ${role === 'user' ? 'fa-user' : 'fa-robot'}"></i></div>
                <div class="bubble">${contentHtml}</div>
            `;
            this.ui.msgContainer.appendChild(div);
            this.scrollToBottom();
        }

        appendLoading(id) {
            const div = document.createElement('div');
            div.id = id;
            div.className = 'message msg-bot';
            div.innerHTML = `
                <div class="avatar"><i class="fas fa-robot"></i></div>
                <div class="bubble" style="color:#94a3b8; font-style:italic;">
                    <i class="fas fa-circle-notch fa-spin"></i> Analizando vectores estrat√©gicos...
                </div>
            `;
            this.ui.msgContainer.appendChild(div);
            this.scrollToBottom();
        }

        saveMessage(role, content) {
            const chat = this.chats.find(c => c.id === this.activeChatId);
            chat.messages.push({ role, content });
            
            // Generar t√≠tulo si es el primer intercambio
            if(chat.messages.length === 2) {
                chat.title = chat.messages[0].content.substring(0, 30) + '...';
                this.ui.titleHeader.textContent = chat.title;
                this.renderHistory();
            }
            
            this.saveData();
        }

        saveData() {
            localStorage.setItem('maxgpt_chats_v2', JSON.stringify(this.chats));
        }

        clearChat() {
            if(confirm('¬øPurgar esta l√≠nea temporal?')) {
                const idx = this.chats.findIndex(c => c.id === this.activeChatId);
                if(idx > -1) {
                    this.chats.splice(idx, 1);
                    this.saveData();
                    this.activeChatId = null;
                    this.renderHistory();
                    this.ui.msgContainer.innerHTML = '';
